\section{Sets}

\begin{definition}[Point]
    A \emph{point} of a set $X$ is a map $1 \rightarrow X$.
\end{definition}

\begin{definition}[Constant]
    A map $f : A \rightarrow B$ is \emph{constant} iff there exists $b : 1 \rightarrow B$ such that $f = b \circ !$.
\end{definition}
